function checkConcreteExists()
        if workspace.Elevator and workspace.Elevator.Ceiling and workspace.Elevator.Ceiling:FindFirstChild("Concrete") then
            return true
        else
            Fluent:Notify({
                Title = "7yd7 | Hub",
                Content = "Concrete ceiling not found in workspace.Elevator.Ceiling!",
                Duration = 8
            })
            return false
        end
    end
    
    function createAFKPart()
        if not checkConcreteExists() then
            return nil
        end
    
        if workspace:FindFirstChild("7yd7") then
            return workspace:FindFirstChild("7yd7")
        end
        
        local concrete = workspace.Elevator.Ceiling:FindFirstChild("Concrete")
        
        local afkPart = Instance.new("Part")
        afkPart.Name = "7yd7"
        afkPart.Size = Vector3.new(21.584325790405273, 7.663055896759033, 17.89038848876953)
        afkPart.Anchored = true
        afkPart.CanCollide = true
        
        local concretePosition = concrete.Position
        afkPart.Position = Vector3.new(concretePosition.X, concretePosition.Y + 5000, concretePosition.Z)
        
        afkPart.Parent = workspace
        
        return afkPart
    end
    
    function isPlayerAbovePart(player, part)
        if not player.Character then return false end
        
        local humanoidRootPart = player.Character:FindFirstChild("HumanoidRootPart")
        if not humanoidRootPart then return false end
        
        local playerPos = humanoidRootPart.Position
        
        local partMin = Vector2.new(part.Position.X - (part.Size.X/2), part.Position.Z - (part.Size.Z/2))
        local partMax = Vector2.new(part.Position.X + (part.Size.X/2), part.Position.Z + (part.Size.Z/2))
        
        if playerPos.X >= partMin.X and playerPos.X <= partMax.X and
           playerPos.Z >= partMin.Y and playerPos.Z <= partMax.Y then
            
            local partTopY = part.Position.Y + (part.Size.Y/2)
            local distanceAbove = playerPos.Y - partTopY
            
            if distanceAbove >= -1 and distanceAbove <= 50 then
                return true
            end
        end
        
        return false
    end

    local function isCoin(obj)
        local function isInsideTool(object)
            local parent = object.Parent
            while parent do
                if parent:IsA("Tool") then return true end
                parent = parent.Parent
            end
            return false
        end
        if obj:IsA("Tool") or isInsideTool(obj) then return false end
        return obj.Name:lower() == "coin"
    end
    
    local function searchForCoins(parent)
        local coins = {}
        local function recursiveSearch(obj)
            if isCoin(obj) then table.insert(coins, obj) end
            for _, child in pairs(obj:GetChildren()) do
                recursiveSearch(child)
            end
        end
        for _, child in pairs(parent:GetChildren()) do
            if not child:IsA("Tool") and child ~= LocalPlayer.Backpack then
                recursiveSearch(child)
            end
        end
        return coins
    end
    
    local function saveOriginalState()
        local camera = workspace.CurrentCamera
        return {
            cframe = HumanoidRootPart.CFrame,
            walkSpeed = Humanoid.WalkSpeed,
            jumpPower = Humanoid.JumpPower,
            hipHeight = Humanoid.HipHeight,
            camSubject = camera.CameraSubject,
            camType = camera.CameraType
        }
    end
    
    local function restoreOriginalState(state)
        if not Character or not Character.Parent then return end
        
        HumanoidRootPart.CFrame = state.cframe
        Humanoid.WalkSpeed = state.walkSpeed
        Humanoid.JumpPower = state.jumpPower
        Humanoid.HipHeight = state.hipHeight
        local camera = workspace.CurrentCamera
        camera.CameraSubject = state.camSubject
        camera.CameraType = state.camType
    end
    
    local function interactWithCoin(coin)
        if not coin or not coin.Parent or not Character or not Character.Parent then return false end
        
        isCollecting = true
        local state = saveOriginalState()
        Humanoid.WalkSpeed = 0
        Humanoid.JumpPower = 0
        workspace.CurrentCamera.CameraType = Enum.CameraType.Scriptable
        HumanoidRootPart.CFrame = typeof(coin.CFrame) == "CFrame" and coin.CFrame or CFrame.new(coin.Position)
        task.wait(interactionDelay)
        local success = (not coin or not coin.Parent)
        
        if Character and Character.Parent then
            restoreOriginalState(state)
        end
        
        isCollecting = false
        return success
    end
    
    local function collectCoinWithRetry(coin)
        if coinAttempts[coin] and coinAttempts[coin] >= maxAttempts then return false end
        coinAttempts[coin] = (coinAttempts[coin] or 0) + 1
        local success = interactWithCoin(coin)
        if success then coinAttempts[coin] = nil end
        return success
    end
    
    local function collectAllCoins()
        for coin, _ in pairs(coinAttempts) do
            if not coin or not coin.Parent then
                coinAttempts[coin] = nil
            end
        end
        local coins = searchForCoins(game.Workspace)
        table.sort(coins, function(a, b)
            return (a.Position - HumanoidRootPart.Position).Magnitude < (b.Position - HumanoidRootPart.Position).Magnitude
        end)
        for _, coin in pairs(coins) do
            if not isCollecting and Character and Character.Parent then
                collectCoinWithRetry(coin)
                task.wait(0.1)
            end
        end
    end
    
    local function coinFarmLoop()
        while autoCoinFarmRunning do
            if Character and Character.Parent and HumanoidRootPart and Humanoid then
                local success, _ = pcall(collectAllCoins)
                if HumanoidRootPart then
                    HumanoidRootPart.CFrame = CFrame.new(HumanoidRootPart.Position)
                end
            end
            task.wait(coinSearchInterval)
        end
    end

                                local function isSeatUsed(seat)
                                    local seatPath = seat:GetFullName()
                                    
                                    for _, usedPath in pairs(getgenv().usedSeatPaths) do
                                        if usedPath == seatPath then
                                            return true
                                        end
                                    end
                                    
                                    if string.match(seat.Name, "%-Y$") then
                                        return true
                                    end
                                    
                                    return false
                                end
                                
                                local function findSeatsInModel(model, fromReplicatedStorage)
                                    local foundSeats = {}
                                    
                                    local function searchInInstance(instance)
                                        for _, child in pairs(instance:GetChildren()) do
                                            if child:IsA("Seat") and child.ClassName == "Seat" then
                                                if not isSeatUsed(child) then
                                                    table.insert(foundSeats, {
                                                        seat = child,
                                                        fromReplicatedStorage = fromReplicatedStorage,
                                                        path = child:GetFullName()
                                                    })
                                                end
                                            end
                                            
                                            searchInInstance(child)
                                        end
                                    end
                                    
                                    searchInInstance(model)
                                    
                                    return foundSeats
                                end

                                    if isElevatorMenuOpen() then
                                    else
                                        return
                                    end
                                    
                                    local seatsInWorkspace = findSeatsInModel(workspace, false)

                                    if #seatsInWorkspace == 0 then
                                        seatsInReplicatedStorage = findSeatsInModel(game:GetService("ReplicatedStorage"), true)
                                    end

                                    for _, seatInfo in pairs(seatsInWorkspace) do
                                        table.insert(allSeats, seatInfo)
                                    end
                                    for _, seatInfo in pairs(seatsInReplicatedStorage) do
                                        table.insert(allSeats, seatInfo)
                                    end

                                        return
                                    end

                                    local selectedSeat = selectedSeatInfo.seat
                                    
                                    local originalCFrame = selectedSeat.CFrame
                                    local originalName = selectedSeat.Name
                                    local originalParent = selectedSeat.Parent
                                    
                                    if selectedSeatInfo.fromReplicatedStorage then
                                        selectedSeat.Parent = workspace
                                    end
                                    
                                    selectedSeat.Name = originalName .. "-Y"
                                    
                                    table.insert(getgenv().usedSeatPaths, selectedSeatInfo.path)

                                    local offsetCFrame = selectedSeat.CFrame * CFrame.new(0, 2, 0) 
                                    HumanoidRootPart.CFrame = offsetCFrame
                                    
                                    local connection

                                                selectedSeat.Parent = originalParent
                                            end
                                            
                                            connection:Disconnect()
                                        end
                                    end)
                                    
                                    wait(0.5)
                                end
                                
                                goToSeatAndUse()
                            end
                        end
                    end
                end
