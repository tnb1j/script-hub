    Multi = true,
    Default = {"All"}
})

function CleanBuildingName(buildingName)
    local cleanName = buildingName:gsub("%d+$", ""):gsub("%s+$", "")
    return cleanName
end

function GetAvailableBuildings()
    local Players = game:GetService("Players")
    local ReplicatedStorage = game:GetService("ReplicatedStorage")
    local LocalPlayer = Players.LocalPlayer
    
    local playerBuildings = LocalPlayer:WaitForChild("Buildings")
    local worldBuildings = workspace.Main.Plot.Buildings:GetChildren()
    
    local availableBuildings = {}
    local cleanNames = {}
    
    for _, building in ipairs(playerBuildings:GetChildren()) do
        local buildingName = building.Name
        
        for _, worldBuilding in ipairs(worldBuildings) do
            if worldBuilding.Name == buildingName then
                if worldBuilding:GetAttribute("Growing") ~= nil then
                    local cleanName = CleanBuildingName(buildingName)
                    
                    table.insert(availableBuildings, buildingName)
                    
                    if not table.find(cleanNames, cleanName) then
                        table.insert(cleanNames, cleanName)
                    end
                end
            end
        end
    end
    
    table.sort(cleanNames)
    table.insert(cleanNames, 1, "All")
    
    getgenv().AvailableBuildings = availableBuildings
    return cleanNames
end

function UpdateBuildingsList()
    spawn(function()
        while true do
            wait(2)
            local newBuildings = GetAvailableBuildings()
            
            local currentValues = {}
            for _, value in ipairs(MultiDropdown.Values or {}) do
                table.insert(currentValues, value)
            end
            
            if #newBuildings ~= #currentValues then
                MultiDropdown:SetValues(newBuildings)
            else
                local isDifferent = false
                for i, building in ipairs(newBuildings) do
                    if building ~= currentValues[i] then
                        isDifferent = true
                        break
                    end
                end
                
                if isDifferent then
                    MultiDropdown:SetValues(newBuildings)
                end
            end
        end
    end)
end

function AutoMineHarvestLoop()
    spawn(function()
        while getgenv().AutoMineHarvests do
            wait(AutoSpeed)
            
            local Players = game:GetService("Players")
            local ReplicatedStorage = game:GetService("ReplicatedStorage")
            local LocalPlayer = Players.LocalPlayer
            
            local KnitServices = ReplicatedStorage.Packages._Index["sleitnick_knit@1.7.0"].knit.Services
            local EventService = KnitServices.EventService
            
            local playerBuildings = LocalPlayer:WaitForChild("Buildings")
            local worldBuildings = workspace.Main.Plot.Buildings:GetChildren()
            
            for _, building in ipairs(playerBuildings:GetChildren()) do
                local buildingName = building.Name
                
                local shouldHarvest = false
                
                if table.find(getgenv().SelectedBuildings, "All") then
                    shouldHarvest = true
                else
                    local cleanName = CleanBuildingName(buildingName)
                    if table.find(getgenv().SelectedBuildings, cleanName) then
                        shouldHarvest = true
                    end
                end
                
                if shouldHarvest then
                    for _, worldBuilding in ipairs(worldBuildings) do
                        if worldBuilding.Name == buildingName then
                            if worldBuilding:GetAttribute("Growing") ~= nil then
                                local ohString1 = buildingName
                                EventService.RF.MineHarvestable:InvokeServer(ohString1)
                            end
                        end
                    end
                end
            end
        end
    end)
end
