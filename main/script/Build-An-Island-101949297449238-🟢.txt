function updateResourcesList()
    local Players = game:GetService("Players")
    local localPlayer = Players.LocalPlayer
    local plotName = localPlayer.Name
    local plot = workspace:FindFirstChild("Plots"):FindFirstChild(plotName)
    
    local newResourcesList = {"All"}
    local resourceNames = {} 
    resourceNames["All"] = true 
    
    if plot and plot:FindFirstChild("Resources") then
        for _, resource in ipairs(plot.Resources:GetChildren()) do
            if resource.Name and not resourceNames[resource.Name] then
                table.insert(newResourcesList, resource.Name)
                resourceNames[resource.Name] = true
            end
        end
    end
    
    local hasChanged = false
    if #newResourcesList ~= #getgenv().ResourcesList then
        hasChanged = true
    else
        for i, resourceName in ipairs(newResourcesList) do
            if getgenv().ResourcesList[i] ~= resourceName then
                hasChanged = true
                break
            end
        end
    end
    
    if hasChanged then
        getgenv().ResourcesList = newResourcesList
        updateDropdownValues()
    end
end

function updateDropdownValues()
    if getgenv().ResourcesDropdown then
        getgenv().ResourcesDropdown:SetValues(getgenv().ResourcesList)
        
        local validSelections = {}
        for _, selectedResource in ipairs(getgenv().SelectedResources) do
            for _, availableResource in ipairs(getgenv().ResourcesList) do
                if selectedResource == availableResource then
                    validSelections[selectedResource] = true
                    break
                end
            end
        end
        
        if next(validSelections) == nil then
            validSelections["All"] = true
            getgenv().SelectedResources = {"All"}
        end
        
        getgenv().ResourcesDropdown:SetValue(validSelections)

    end
end

function shouldHitResource(resourceName)
    for _, selected in ipairs(getgenv().SelectedResources or {}) do
        if selected == "All" then
            return true
        end
        if selected == resourceName then
            return true
        end
    end
    return false
end

function isResourceFullyGrown(resource)
    local highestNumber = -1
    
    for _, child in ipairs(resource:GetChildren()) do
        local number = tonumber(child.Name)
        if number and number > highestNumber then
            highestNumber = number
        end
    end
    
    if highestNumber >= 0 then
        local highestChild = resource:FindFirstChild(tostring(highestNumber))
        if highestChild then
            for _, part in ipairs(highestChild:GetChildren()) do
                if part:IsA("Part") then
                    return part.Transparency == 0
                end
            end
        end
    end
    
    return false
end

function hitWhenFullyGrown(resource)
    local numbers = {}
    for _, child in ipairs(resource:GetChildren()) do
        local number = tonumber(child.Name)
        if number and number >= 0 then
            table.insert(numbers, number)
        end
    end
    
    if #numbers == 0 then return false end
    
    table.sort(numbers, function(a, b) return a > b end)
    
    local highestNumber = numbers[1]
    local highestChild = resource:FindFirstChild(tostring(highestNumber))
    if not highestChild then return false end
    
    local highestPart = nil
    for _, part in ipairs(highestChild:GetChildren()) do
        if part:IsA("Part") then
            highestPart = part
            break
        end
    end
    
    if not highestPart or highestPart.Transparency ~= 0 then
        return false 
    end
    
    for _, number in ipairs(numbers) do
        local numberChild = resource:FindFirstChild(tostring(number))
        if numberChild then
            for _, part in ipairs(numberChild:GetChildren()) do
                if part:IsA("Part") then
                    if part.Transparency == 1 then
                        return true
                    elseif part.Transparency == 0 then
                        local success, errorMsg = pcall(function()
                            game:GetService("ReplicatedStorage").Communication.HitResource:FireServer(resource)
                        end)
                    end
                    break 
                end
            end
        end
    end
    
    return true
end

                for _, resource in ipairs(plot.Resources:GetChildren()) do
                    if shouldHitResource(resource.Name) then
                        if resource and resource.Parent then
                            if getgenv().HitMode == "Hit When Growing" then
                                local success, errorMsg = pcall(function()
                                    ReplicatedStorage.Communication.HitResource:FireServer(resource)
                                end)
                            elseif getgenv().HitMode == "Hit When Fully Grown" then
                                hitWhenFullyGrown(resource)
                            end
                        end
                    end
                end
            end
        end
    end)
end

local localPlayer = Players.LocalPlayer
local plotName = localPlayer.Name
local expandFolder = workspace:WaitForChild("Plots"):FindFirstChild(plotName):FindFirstChild("Expand")
local currenciesFolder = localPlayer:WaitForChild("PlayerGui"):WaitForChild("Main")
    :WaitForChild("Left"):WaitForChild("Currencies")

for _, expandItem in ipairs(expandFolder:GetChildren()) do
    local ohString1 = expandItem.Name
    local billboardGui = expandItem:FindFirstChild("Top") and expandItem.Top:FindFirstChild("BillboardGui")
    if billboardGui then
        for _, guiElement in ipairs(billboardGui:GetChildren()) do
            if guiElement:IsA("Frame") and guiElement.Name ~= "UIListLayout" and guiElement.Name ~= "Example" then
                local ohString2 = guiElement.Name
                local ohNumber3 = 1
                
                if guiElement:FindFirstChild("Amount") then
                    local amountText = guiElement.Amount.Text
                    local maxAmount = string.match(amountText, "/(%d+)")
                    if maxAmount then
                        ohNumber3 = tonumber(maxAmount)
                    end
                end
                
                local currencyItem = currenciesFolder:FindFirstChild(ohString2)
                if currencyItem and currencyItem:FindFirstChild("Display") then
                    local function parseShortNumber(text)
                        text = string.lower(text)
                        local number = tonumber(text)
                        if number then
                            return number
                        end
                        local value = tonumber(string.match(text, "[%d%.]+"))
                        local suffix = string.match(text, "[a-z]+")
                        if value and suffix then
                            if suffix == "k" then
                                return value * 1_000
                            elseif suffix == "m" then
                                return value * 1_000_000
                            elseif suffix == "b" then
                                return value * 1_000_000_000
                            elseif suffix == "t" then
                                return value * 1_000_000_000_000
                            end
                        end
                        return nil
                    end
                    
                    local displayText = currencyItem.Display.Text
                    local amount = parseShortNumber(displayText)
                    if amount and amount > 1 then
                        ReplicatedStorage.Communication.ContributeToExpand:FireServer(ohString1, ohString2, ohNumber3)
                    end
                end
            end
        end
    end
end
end
    end)
end

function updateCraftsList()
    local Players = game:GetService("Players")
    local localPlayer = Players.LocalPlayer
    local plotName = localPlayer.Name
    
    if not workspace:FindFirstChild("Plots") then
        return
    end
    
    local plot = workspace.Plots:FindFirstChild(plotName)
    if not plot then
        return
    end
    
    local landFolder = plot:FindFirstChild("Land")
    if not landFolder then
        return
    end
    
    local newCraftsList = {"All"}
    local craftNames = {}
    craftNames["All"] = true
    
    for _, item in ipairs(landFolder:GetChildren()) do
        local crafter = item:FindFirstChild("Crafter")
        if crafter then
            local attachment = crafter:FindFirstChild("Attachment")
            if attachment then
                local item1 = attachment:GetAttribute("Item1") or attachment:GetAttribute("item1")
                local item2 = attachment:GetAttribute("Item2") or attachment:GetAttribute("item2")
                local output = attachment:GetAttribute("Output") or attachment:GetAttribute("output")
                
                local craftName = ""
                
                if output then
                    if item1 and item2 then
                        craftName = item1 .. " + " .. item2 .. " > " .. output
                    elseif item1 then
                        craftName = item1 .. " > " .. output
                    end
                else
                    if item1 and item2 then
                        craftName = item1 .. " > " .. item2
                    end
                end
                
                if craftName ~= "" and not craftNames[craftName] then
                    table.insert(newCraftsList, craftName)
                    craftNames[craftName] = true
                end
            end
        end
    end
    
    local hasChanged = false
    if #newCraftsList ~= #getgenv().CraftsList then
        hasChanged = true
    else
        for i, craftName in ipairs(newCraftsList) do
            if getgenv().CraftsList[i] ~= craftName then
                hasChanged = true
                break
            end
        end
    end
    
    if hasChanged then
        getgenv().CraftsList = newCraftsList
        updateDropdownValues()
    end
end

local lastUpdateTime = 0
local cachedValues = {}
local isUpdating = false

local function UpdateMerchantDropdown()
    if isUpdating then return end 
    isUpdating = true
    
    local currentTime = tick()
    local rawItems = {}
    
    local success, result = pcall(function()
        local children = merchantPath:GetChildren()
        
        for i = 1, #children do
            local child = children[i]
            local className = child.ClassName
            
            if className ~= "UIAspectRatioConstraint" and
               className ~= "UIGridLayout" and
               className ~= "UIPadding" and
               child.Name ~= "Example" then
                
                local count = 0
                local isVisible = child.Visible
                
                if isVisible then
                    local countObj = child:FindFirstChild("Count")
                    
                    if countObj and countObj.ClassName == "TextLabel" and countObj.Visible then
                        local text = countObj.Text
                        if text and text ~= "" then
                            local match = string.match(text, "x(%d+)")
                            count = match and tonumber(match) or 0
                        end
                    end
                else
                    count = 0 
                end
                
                rawItems[#rawItems + 1] = {
                    Name = child.Name,
                    Count = count,
                    Visible = isVisible
                }
            end
        end
        
        table.sort(rawItems, function(a, b)
            return a.Count > b.Count
        end)
        
        return rawItems
    end)
    
    if success and result then
        local dropdownValues = {"All"}
        local newNameToRealName = {}
        
        for i = 1, #result do
            local item = result[i]
            local display = item.Name
            
            if item.Count > 0 and item.Visible then
                display = display .. " | x" .. item.Count
            elseif not item.Visible then
                display = display
            else
                display = display .. " | x0"
            end
            
            dropdownValues[#dropdownValues + 1] = display
            newNameToRealName[display] = item.Name
        end
        
        local hasChanged = false
        if #cachedValues ~= #dropdownValues then
            hasChanged = true
        else
            for i = 1, #dropdownValues do
                if cachedValues[i] ~= dropdownValues[i] then
                    hasChanged = true
                    break
                end
            end
        end
        
        if hasChanged then
            cachedValues = dropdownValues
            nameToRealName = newNameToRealName
            
            task.defer(function()
                Dropdown:SetValues(dropdownValues)
            end)
        end
    end
    
    lastUpdateTime = currentTime
    isUpdating = false
end

                        ReplicatedStorage.Communication.BuyFromMerchant:FireServer(name, false)
                    end)
                end
            end
            
            local buyAll = false
            if type(selectedValues) == "table" then
                for _, value in pairs(selectedValues) do
                    if value == "All" then
                        buyAll = true
                        break
                    end

            end
            
            if buyAll then
                local itemsToFire = {}
                for display, realName in pairs(nameToRealName) do
                    local countMatch = string.match(display, "| x(%d+)")
                    if countMatch and not string.find(display, "Hidden") then
                        local count = tonumber(countMatch)
                        if count and count > 0 then
                            table.insert(itemsToFire, realName)
                        end
                    end
                end
                
                for i = 1, #itemsToFire do
                    if not getgenv().AutoBuyFromMerchantA then break end
                    fire(itemsToFire[i])
                    if i < #itemsToFire then
                        task.wait(0.05)
                    end
                end
            else
                local itemsToProcess = {}
                
                if type(selectedValues) == "table" then
                    itemsToProcess = selectedValues
                else
                    itemsToProcess = {selectedValues}
                end
                
                for _, selected in pairs(itemsToProcess) do
                    if selected ~= "All" then
                        local realName = nameToRealName[selected]
                        if realName and not string.find(selected, "Hidden") then
                            local countMatch = string.match(selected, "| x(%d+)")
                            if countMatch then
                                local count = tonumber(countMatch)
                                if count and count > 0 then
                                    fire(realName)
                                end
                            elseif not string.find(selected, "| x") then
                                fire(realName)
                            end
                        end
                    end
                end
            end
        end
    end)
end

local function parseGoldAmount(text)
    if not text then return 0 end
    
    text = text:gsub("Gold", ""):gsub("%s+", "")
    
    local number = text:match("([%d%.]+)")
    if not number then return 0 end
    
    number = tonumber(number)
    if not number then return 0 end
    
    if text:lower():find("k") then
        number = number * 1000
    elseif text:lower():find("m") then
        number = number * 1000000
    elseif text:lower():find("b") then
        number = number * 1000000000
    end
    
    return math.floor(number)
end

local function getUpgradeCost(upgradeName)
    local player = game:GetService("Players").LocalPlayer
    local upgradeButton = player.PlayerGui.Main.Menus.GoldUpgrades.Inner.ScrollingFrame.Hold:FindFirstChild(upgradeName)
    
    if upgradeButton and upgradeButton:FindFirstChild("Button") and upgradeButton.Button:FindFirstChild("TextLabel") then
        local costText = upgradeButton.Button.TextLabel.Text
        return parseGoldAmount(costText)
    end
    
    return math.huge 
end

local function getCurrentGold()
    local player = game:GetService("Players").LocalPlayer
    if player.leaderstats and player.leaderstats:FindFirstChild("Gold") then
        return player.leaderstats.Gold.Value
    end
    return 0
end

local function findCheapestAffordableUpgrade()
    local currentGold = getCurrentGold()
    local cheapestUpgrade = nil
    local cheapestCost = math.huge
    
    for _, upgradeName in pairs(getgenv().SelectedUpgrades or {}) do
        local cost = getUpgradeCost(upgradeName)
        
        if cost <= currentGold and cost < cheapestCost then
            cheapestUpgrade = upgradeName
            cheapestCost = cost
        end
    end
    
    return cheapestUpgrade, cheapestCost
end
